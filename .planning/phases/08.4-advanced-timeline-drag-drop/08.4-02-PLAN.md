```typescript
const detectInsertPosition = useCallback((
  trackId: string,
  timePosition: number,
  excludeClipId: string
): { index: number; snapTime: number } | null => {
  const state = useTimelineStore.getState();
  const clipIds = (state.trackClips[trackId] || []).filter(id => id !== excludeClipId);
  
  // Duyệt gap giữa các clips, tìm gap gần nhất
  for (let i = 0; i <= clipIds.length; i++) {
    const prevClip = i > 0 ? state.clips[clipIds[i - 1]] : null;
    const nextClip = i < clipIds.length ? state.clips[clipIds[i]] : null;
    
    const gapStart = prevClip ? prevClip.timelineStart + prevClip.timelineDuration : 0;
    const gapEnd = nextClip ? nextClip.timelineStart : Infinity;
    
    // Nếu mouse position nằm gần boundary (±threshold)
    const threshold = 0.5; // giây
    if (Math.abs(timePosition - gapStart) < threshold || 
        (timePosition >= gapStart && timePosition <= gapEnd)) {
      return { index: i, snapTime: gapStart };
    }
  }
  return null;
}, []);
```

3. Export tất cả state và functions mới.

2. Implement `rippleInsert`:
```typescript
rippleInsert: (clipId, targetTrackId, insertIndex) => set((state) => {
  const clip = state.clips[clipId];
  if (!clip) return state;
  
  const newClips = { ...state.clips };
  const oldTrackId = clip.trackId;
  
  // Xóa clip khỏi track cũ
  const oldTrackClips = (state.trackClips[oldTrackId] || []).filter(id => id !== clipId);
  
  // Tính timelineStart mới = end time của clip trước insertIndex
  let targetTrackClips = [...(state.trackClips[targetTrackId] || [])];
  let newStart = 0;
  if (insertIndex > 0 && targetTrackClips.length > 0) {
    const prevClipId = targetTrackClips[Math.min(insertIndex - 1, targetTrackClips.length - 1)];
    const prevClip = newClips[prevClipId];
    if (prevClip) newStart = prevClip.timelineStart + prevClip.timelineDuration;
  }
  
  // Update clip trackId & position
  newClips[clipId] = { ...clip, trackId: targetTrackId, timelineStart: newStart };
  
  // Dồn tất cả clips SAU insertIndex ra phía sau
  const insertedDuration = clip.timelineDuration;
  for (let i = insertIndex; i < targetTrackClips.length; i++) {
    const cId = targetTrackClips[i];
    const c = newClips[cId];
    if (c) newClips[cId] = { ...c, timelineStart: c.timelineStart + insertedDuration };
  }
  
  // Insert clipId vào đúng vị trí
  targetTrackClips.splice(insertIndex, 0, clipId);
  
  return {
    clips: newClips,
    trackClips: {
      ...state.trackClips,
      [oldTrackId]: oldTrackClips,
      [targetTrackId]: targetTrackClips,
    },
  };
}),
```
</action>
