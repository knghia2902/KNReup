```typescript
useEffect(() => {
  const handleKeyDown = (e: KeyboardEvent) => {
    if (['INPUT', 'TEXTAREA'].includes((e.target as HTMLElement).tagName)) return;
    const cmdOrCtrl = e.metaKey || e.ctrlKey;
    
    // Split at playhead (Ctrl+B)
    if (cmdOrCtrl && e.key.toLowerCase() === 'b') {
      // TODO: implement split cho clip model trong phase sau
      e.preventDefault();
    }
    
    // Delete (Del/Backspace)
    if (e.key === 'Delete' || e.key === 'Backspace') {
      const { selectedClipId } = useTimelineStore.getState();
      if (selectedClipId) {
        if (e.shiftKey) {
          useTimelineStore.getState().rippleDelete(selectedClipId); // D-08
        } else {
          useTimelineStore.getState().removeClip(selectedClipId);
        }
        e.preventDefault();
      }
    }
  };
  window.addEventListener('keydown', handleKeyDown);
  return () => window.removeEventListener('keydown', handleKeyDown);
}, [currentTime]);
```

// Timeline.tsx: onDrop handler on the tlbody container
const handleDrop = (e: React.DragEvent) => {
  e.preventDefault();
  const data = e.dataTransfer.getData('application/json');
  if (!data) return;
  
  const { filePath, mediaType } = JSON.parse(data);
  
  if (mediaType === 'video') {
    // D-10: Append video vào Main Track
    useTimelineStore.getState().appendClipToTrack('main', {
      type: 'video',
      sourceFile: filePath,
      sourceStart: 0,
      sourceDuration: 0, // Will be set after metadata loaded
      timelineDuration: 0,
    });

    // D-11: Append audio vào BGM Track
    useTimelineStore.getState().appendClipToTrack('bgm', {
      type: 'audio',
      sourceFile: filePath,
      sourceStart: 0,
      sourceDuration: 0,
      timelineDuration: 0,
    });
  }
};

**1. Playhead sync effect (giữ nguyên logic, update sources):**

```typescript
useEffect(() => {
  let animationFrameId: number;
  let lastTime = -1;
  const updatePlayhead = () => {
    if (playheadRef.current && !drag.isDraggingPlayhead) {
      const video = document.querySelector('video');
      if (video) {
        const time = video.currentTime;
        if (time !== lastTime) {
          lastTime = time;
          setCurrentTime(time);
          const xPos = time * pixelsPerSecond;
          playheadRef.current.style.transform = `translateX(${xPos + TIMELINE_OFFSET_X - scrollLeft}px)`;
        }
      }
    }
    animationFrameId = requestAnimationFrame(updatePlayhead);
  };
  animationFrameId = requestAnimationFrame(updatePlayhead);
  return () => cancelAnimationFrame(animationFrameId);
}, [pixelsPerSecond, drag.isDraggingPlayhead, scrollLeft]);
```
